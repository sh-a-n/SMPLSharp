446  public void Schedule(int event_id, int wait_time, object param = null)
447  {
448  var e = new SmplEvent(event_id, Time, wait_time, param);
449  for (int i = futureEvents.Count - 1; i >= 0; i--)

452  {

476  public bool Cause(out SmplEvent model_event)

551  public int Cancel(int event_id, object param = null)
552  {
553  foreach (SmplEvent e in futureEvents)
554  {
555  if ((e.EventID == event_id) && (e.Param == param))
556  {
557  int ind = futureEvents.IndexOf(e);
558  futureEvents.RemoveAt(ind);
559  return e.TimeCaused - this.Time;
560  }
